local CanDrag = true
local dragging

function CreateDrag(gui)
	local dragInput
	local dragStart
	local startPos

	function update(input)
		local delta = input.Position - dragStart
		TS:Create(gui, TweenInfo.new(0.16, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)}):Play();
	end

	local lastEnd = 0
	local lastMoved = 0
	local con
	gui.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			if not CanDrag then return end
			dragging = true
			dragStart = input.Position
			startPos = gui.Position
		end
	end)

	UIS.InputEnded:Connect(function(input)

		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = false
		end
	end)


	gui.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
			lastMoved = os.clock()
		end
	end)

	UIS.InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			update(input)
		end
	end)
end

		if isDragging and input.UserInputType == Enum.UserInputType.MouseMovement then
			local mousePos = UIS:GetMouseLocation()
			local delta = mousePos - initialMousePos
			local newSize = UDim2.new(
				initialSize.X.Scale,
				initialSize.X.Offset + delta.X,
				initialSize.Y.Scale,
				initialSize.Y.Offset + delta.Y
			)
			Main.Size = newSize
		end
	end)

	UIS.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then

		end
	end)

function saveConfig(FileName: string)
	if not isfile("Saturn") then
		makefolder("Saturn")
	end

	local Settings = {}
	for i, v in next, Flags do
		warn(v.Component)
		if v.Component == "Toggle" then
			table.insert(Settings, {
				["Name"]        = v.Name,
				["Value"]       = v.Value,
				["Component"] = "Toggle"
			})
		elseif v.Component == "Slider" then
			table.insert(Settings, {
				["Name"]        = v.Name,
				["Value"]       = v.Value,
				["Component"] = "Slider"
			})
		elseif v.Component == "Keybind" and v.Value ~= nil then
			table.insert(Settings, {
				["Name"]      = v.Name,
				["Value"]      = tostring(v.Value),
				["Component"] = "Keybind"
			})
		elseif v.Component == "Dropdown" then
			table.insert(Settings, {
				["Name"]      = v.Name,
				["Value"]      = v.Default,
				["Component"] = "Dropdown"
			})
		elseif v.Component == "Colorpicker" then
			table.insert(Settings, {
				["Name"]      = v.Name,
				["RValue"]      = v.Value.R * 255,
				["GValue"]    = v.Value.G * 255,
				["BValue"]    = v.Value.B * 255,
				["Transp"]    = v.Transparency,
				["Component"] = "Colorpicker"
			})
		end
	end

	local Encoded = game:GetService("HttpService"):JSONEncode(Settings)
	local FileName = "Saturn\\"..FileName..".saturn"
	writefile(FileName, "")
	appendfile(FileName, Encoded)
end

function followBall()
	if not Exceros.Catching.HitboxEnabled then return end

	local closestBall = nil
	local closestDistance = math.huge

	for _, v in pairs(Workspace:GetChildren()) do
		if v.Name == "Football" and v:IsA("BasePart") and not v.Anchored then
			local distance = (char.HumanoidRootPart.Position - v.Position).Magnitude
			if distance < closestDistance then
				closestDistance = distance
				closestBall = v
			end
		end
	end

	if closestBall then
		hitbox.CFrame = closestBall.CFrame
		hitbox.Size = Vector3.new(
			Exceros.Catching.CustomizableRadius,
			Exceros.Catching.CustomizableRadius,
			Exceros.Catching.CustomizableRadius
		)
	end
end

function updateMagnetBehavior()
	for football, _ in pairs(footballs) do
		footballs[football] = nil
	end

	if Exceros.Catching.Mode == "Blatant" then
		function createFootballHandler(football)
			local state = { hasChangedSize = false }
			function setFootballSize()
				if Exceros.Catching.Magnets and not state.hasChangedSize then
					if football:IsA("Part") then
						pcall(function()
							football.Size = Vector3.new(
								Exceros.Catching.CustomizableRadius,
								Exceros.Catching.CustomizableRadius,
								Exceros.Catching.CustomizableRadius
							)
							football.CanCollide = false
						end)
					end
					state.hasChangedSize = true
				end
			end
			setFootballSize()
			return { setFootballSize = setFootballSize }
		end

		Workspace.ChildAdded:Connect(function(child)
			if child.Name == "Football" then
				local handler = createFootballHandler(child)
				footballs[child] = handler
				child.AncestryChanged:Connect(function(_, parent)
					if not parent then
						footballs[child] = nil
					end
				end)
			end
		end)

	elseif Exceros.Catching.Mode == "Legit" then
		function findClosestFootball()
			local minDistance = math.huge
			local closestFootball = nil
			for _, item in next, Workspace:GetChildren() do
				if item.Name ~= "Football" or not item:IsA("BasePart") then
					continue
				end
				local rootPart = char:FindFirstChild("HumanoidRootPart")
				if not rootPart then
					continue
				end
				local currentDistance = (item.Position - rootPart.Position).Magnitude
				if currentDistance < minDistance then
					closestFootball = item
					minDistance = currentDistance
				end
			end
			return closestFootball
		end

		task.spawn(function()
			while Exceros.Catching.Magnets do
				task.wait(Exceros.Catching.CustomizableDelay)
				local ball = findClosestFootball()
				if not ball then
					continue
				end
				local catchRight = char:FindFirstChild("CatchRight")
				if not catchRight then continue end
				local distance = (catchRight.Position - ball.Position).Magnitude
				if distance < Exceros.Catching.CustomizableRadius then
					ball.Position = catchRight.Position
				end
			end
		end)
	end
end

function moveToFootball()
    local closestFootball, distance = getClosestFootball()    
    if closestFootball and distance <= pullRangeValue and pullEnabled then        
        if closestFootball.Position.Y > minFootballHeight then            
            local direction = (closestFootball.Position - humanoidRootPart.Position).Unit
            local speed = pullSpeedValue * 10
            humanoidRootPart.Velocity = direction * speed 
            
            if closestFootball.Parent then                
                closestFootball:GetPropertyChangedSignal("Parent"):Connect(function()
                    if closestFootball.Parent and closestFootball.Parent:IsA("Tool") then
                        print("Football picked up, stopping movement")
                        humanoidRootPart.Velocity = Vector3.new(0, 0, 0)
                    end
                end)
            end
        end
    end
end

function followFootball()
    local closestFootball, distance = getClosestFootball()
    if closestFootball and distance <= teleportRangeValue and teleportEnabled then
        if closestFootball.Position.Y > minFootballHeight then
            task.wait(teleportDelay)
            local newPosition = closestFootball.Position + Vector3.new(0, 2, 0)
            character:SetPrimaryPartCFrame(CFrame.new(newPosition))
        end
    end
end

function updateNearestFootball()
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    nearestFootball = nil
    local closestDist = math.huge

    for _, obj in pairs(ws:GetChildren()) do
        if obj.Name == "Football" and obj:IsA("BasePart") then
            local dist = (obj.Position - hrp.Position).Magnitude
            if dist <= pvdist and dist < closestDist then
                closestDist = dist
                nearestFootball = obj
            end
        end
    end
end

function getPredictedPosition(football)
    local futurePosition = football.Position + (football.Velocity * predictionMultiplier)
    return futurePosition
end

function pullVec()
    if pvon and nearestFootball then
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if hrp then
            local predictedPosition = getPredictedPosition(nearestFootball)

            if nearestFootball.Position.Y > hrp.Position.Y - groundThreshold and predictedPosition.Y > hrp.Position.Y then
                local dir = (predictedPosition - hrp.Position).Unit
                hrp.Velocity = dir * pvstr
            end
        end
    end
end

function downForce()
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if hrp then
        if hrp.Velocity.Y > 0 then
            hrp.Velocity = Vector3.new(hrp.Velocity.X, 0, hrp.Velocity.Z)
        end
        
        hrp.Velocity = Vector3.new(hrp.Velocity.X, -dforce * 5, hrp.Velocity.Z)
        
        spawn(function()
            for i = 1, 5 do
                if hrp and hrp.Parent then
                    hrp.Velocity = Vector3.new(hrp.Velocity.X, -dforce * 5, hrp.Velocity.Z)
                    hrp.AssemblyLinearVelocity = Vector3.new(hrp.AssemblyLinearVelocity.X, -dforce * 5, hrp.AssemblyLinearVelocity.Z)
                end
                task.wait(0.01)
            end
        end)
    end
end

local isAntiJamEnabled = false

function updateCollisionState()
	while true do
		if isAntiJamEnabled then
			if game:GetService("Players").LocalPlayer.Character and game:GetService("Players").LocalPlayer.Character:FindFirstChild("Head") and game:GetService("Players").LocalPlayer.Character.Head.CanCollide then
				for _, Player in ipairs(game:GetService("Players"):GetPlayers()) do
					if Player ~= game:GetService("Players").LocalPlayer and Player.Character and Player.Character:FindFirstChild("Head") then
						pcall(function()
							Player.Character.Torso.CanCollide = false
							Player.Character.Head.CanCollide = false
						end)
					end
				end
			end
		else
			if game:GetService("Players").LocalPlayer.Character and game:GetService("Players").LocalPlayer.Character:FindFirstChild("Head") and not game:GetService("Players").LocalPlayer.Character.Head.CanCollide then
				game:GetService("Players").LocalPlayer.Character.Torso.CanCollide = true
				game:GetService("Players").LocalPlayer.Character.Head.CanCollide = true
			end
		end
		task.wait()
	end
end

function Grapher:CreateBeamSegment(startPos, endPos)
    local hiddenFolder = Grapher:CreateHiddenFolder()

    if Grapher.Mode == "Original" then
        local segmentHolder = Grapher.BeamHolderTemplate:Clone()
        segmentHolder.Parent = hiddenFolder

        local attachment0 = Instance.new("Attachment", segmentHolder)
        attachment0.WorldPosition = startPos

        local attachment1 = Instance.new("Attachment", segmentHolder)
        attachment1.WorldPosition = endPos

        local beam = Instance.new("Beam")
        beam.Attachment0 = attachment0
        beam.Attachment1 = attachment1
        beam.Color = ColorSequence.new(Grapher.BeamColor)
        beam.Transparency = NumberSequence.new(Grapher.BeamTransparency)
        beam.Width0 = 0.2
        beam.Width1 = 0.2
        beam.Parent = segmentHolder

        task.delay(4, function()
            if segmentHolder then
                segmentHolder:Destroy()
            end
        end)
    elseif Grapher.Mode == "Saber" then
        local saberPart = Instance.new("Part")
        saberPart.Anchored = true
        saberPart.CanCollide = false
        saberPart.Size = Vector3.new(0.5, 0.5, (startPos - endPos).Magnitude)
        saberPart.CFrame = CFrame.new((startPos + endPos) / 2, endPos)
        saberPart.BrickColor = BrickColor.new(Grapher.BeamColor)
        saberPart.Material = Enum.Material.Neon
        saberPart.Parent = hiddenFolder

        local glow = Instance.new("SurfaceLight")
        glow.Angle = 180
        glow.Brightness = 5
        glow.Range = 4
        glow.Color = Grapher.BeamColor
        glow.Parent = saberPart

        local texture = Instance.new("Texture")
        texture.Texture = "rbxassetid://1243374081"
        texture.StudsPerTileU = 1
        texture.StudsPerTileV = 0.5
        texture.Parent = saberPart

        task.delay(4, function()
            if saberPart then
                saberPart:Destroy()
            end
        end)
    elseif Grapher.Mode == "Beam" then
        local segmentHolder = Grapher.BeamHolderTemplate:Clone()
        segmentHolder.Parent = hiddenFolder

        local attachmentStart = Instance.new("Attachment", segmentHolder)
        attachmentStart.WorldPosition = startPos

        local attachmentEnd = Instance.new("Attachment", segmentHolder)
        attachmentEnd.WorldPosition = endPos

        local beam = Instance.new("Beam")
        beam.Attachment0 = attachmentStart
        beam.Attachment1 = attachmentEnd
        beam.Color = ColorSequence.new(Grapher.BeamColor)
        beam.Transparency = NumberSequence.new(Grapher.BeamTransparency)
        beam.Width0 = 1
        beam.Width1 = 0.5
        beam.LightEmission = 0.5
        beam.Texture = "rbxassetid://121403057"
        beam.TextureLength = 2
        beam.TextureSpeed = 4
        beam.FaceCamera = true
        beam.Parent = segmentHolder

        task.delay(4, function()
            if segmentHolder then
                segmentHolder:Destroy()
            end
        end)
    end
end

function Grapher:GetTrajectoryPoints(origin, velocity)
    local elapsed = 0
    local points = {}

    while elapsed < 5 do
        elapsed = elapsed + Grapher.CastStep
        local nextPos = origin + velocity * elapsed - Vector3.new(0, 0.5 * 28 * elapsed ^ 2, 0)
        table.insert(points, nextPos)
    end

    return points
end

function Grapher:UpdateBeam(points)
    if #points < 2 then return end

    for i = 1, #points - 1 do
        Grapher:CreateBeamSegment(points[i], points[i + 1])
    end
end

function Grapher:GetLanding(origin, velocity, target)
    local trajectoryPoints = Grapher:GetTrajectoryPoints(origin, velocity)
    Grapher:UpdateBeam(trajectoryPoints)
end

function getNearestBall(checkFunc)
	local lowestDistance = math.huge
	local lowestFB = nil
	for index, part in pairs(workspace:GetChildren()) do
		if isFootball(part) and not part.Anchored then
			if checkFunc then
				if not checkFunc(part) then
					continue
				end
			end
			local distance = (Player.Character.HumanoidRootPart.Position - part.Position).Magnitude
			if distance < lowestDistance then
				lowestFB = part
				lowestDistance = distance
			end
		end
	end
	return lowestFB, lowestDistance
end

function getNearestPartToPartFromParts(parts, part)
	local lowestMagnitude = math.huge
	local lowestPart = nil
	for index, p in pairs(parts) do
		local dis = (part.Position - p.Position).Magnitude
		if dis < lowestMagnitude then
			lowestMagnitude = dis
			lowestPart = p
		end
	end
	return lowestPart
end

				firetouchinterest(ball, catch, 0)
				firetouchinterest(ball, catch, 1)
			end
		end
	end
end

function createInstance(className, properties)
    local instance = Instance.new(className)
    for k, v in pairs(properties) do
        if typeof(k) ~= 'string' then
            continue
        end
        instance[k] = v
    end
    return instance
end

function onChatted(p, msg)
    local hidden = true

    if p == Player and msg:lower():sub(1, 4) == "/spy" then
        return
    end

    msg = msg:gsub("[\n\r]", ''):gsub("\t", ' '):gsub("[ ]+", ' ')

    if not spyEnabled then return end

    local processed = false
    local messageId = p.UserId .. msg

    local conn
    conn = getmsg.OnClientEvent:Connect(function(packet, channel)
        if processed then
            return
        end

        if packet.SpeakerUserId == p.UserId and packet.Message == msg:sub(#msg - #packet.Message + 1) and 
           (channel == "All" or (channel == "Team" and not Config.public and Players[packet.FromSpeaker].Team == Player.Team)) then
            hidden = false
            processed = true
        end
    end)

    task.wait(1)

    conn:Disconnect()

    if hidden and not processed then
        if Config.mode == "Chat" then
            if Config.public then
                saymsg:FireServer((Config.publicItalics and "/me " or '') .. "{Saturn Spy} [" .. p.Name .. "]: " .. msg, "All")
            else
                PrivateProperties.Text = "{Saturn Spy} [" .. p.Name .. "]: " .. msg
                StarterGui:SetCore("ChatMakeSystemMessage", PrivateProperties)
            end
        elseif Config.mode == "Notification" then
            showNotification("{Saturn Spy} [" .. p.Name .. "]: " .. msg)
        end
    end
end

function interpolateColor(color1, color2, fraction)
	return Color3.new(
		color1.R + (color2.R - color1.R) * fraction,
		color1.G + (color2.G - color1.G) * fraction,
		color1.B + (color2.B - color1.B) * fraction
	)
end

function cycRain(cols, cb)
	while isCyc or isRedTheme do
		for i = 1, #cols do
			if not isCyc and not isRedTheme then return end
			
			local from = cols[i]
			local to = cols[(i % #cols) + 1]

			for j = 0, 1, 0.1 do
				local newColor = interpolateColor(from, to, j)
				cb(newColor)
				task.wait(0.05)
			end
		end
	end
end

	stopThemeCycle = true

		cycRain(cols, function(color)
			if stopThemeCycle then return end
			Lib:UpdateTheme({

			})
		end)
	end)
end
